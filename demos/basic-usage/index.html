// Basic Usage Example
const { createMonacoRecorder } = window.monacoRecorder;

// Create editor instance
const editor = monaco.editor.create(
  document.getElementById('editor'),
  { value: '// Start recording...\n' }
);

// Initialize recorder
const recorder = createMonacoRecorder(editor);

// Load and play recording
fetch('recording.json')
  .then(response => response.json())
  .then(events => {
    recorder.load(events);
    document.getElementById('playBtn').addEventListener('click', () => {
      recorder.play();
    });
  });
